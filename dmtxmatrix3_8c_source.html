312extern DmtxPassFail dmtxMatrix3VMultiply(OUT DmtxVector2 *vOut, DmtxVector2 *vIn, DmtxMatrix3 m)
313{
314 double w;
315
316 w = vIn->x * m[0][2] + vIn->y * m[1][2] + m[2][2];
317 if (fabs(w) <= DmtxAlmostZero) {
318 vOut->x = FLT_MAX;
319 vOut->y = FLT_MAX;
320 return DmtxFail;
321 }
322
323 vOut->x = (vIn->x * m[0][0] + vIn->y * m[1][0] + m[2][0]) / w;
324 vOut->y = (vIn->x * m[0][1] + vIn->y * m[1][1] + m[2][1]) / w;
325
326 return DmtxPass;
327}
